    <script>
        const num1 = document.getElementById('num1');
        const num2 = document.getElementById('num2');
        const operation = document.getElementById('operation');
        const result = document.getElementById('result');
        const count = document.getElementById('count');
        let calcCount = 0;
        const specialResult = 'Craciun'.length + 5;

        function calculate() {
            let n1 = parseFloat(num1.value);
            let n2 = parseFloat(num2.value);
            let res = 0;

            switch (operation.value) {
                case '+':
                    res = n1 + n2;
                    break;
                case '-':
                    res = n1 - n2;
                    break;
                case '*':
                    res = n1 * n2;
                    break;
                case '/':
                    res = n1 / n2;
                    break;
                case 'sqrt':
                    res = Math.sqrt(n1);
                    break;
                case 'pow':
                    res = Math.pow(n1, n2);
                    break;
                default:
                    res = 0;
            }

            result.textContent = res;

            if (res === specialResult) {
                alert('Ana are mere!');
            } else {
                calcCount++;
                count.textContent = calcCount;
            }
        }

        num1.addEventListener('input', calculate);
        num2.addEventListener('input', calculate);
        operation.addEventListener('change', calculate);
    </script>
